# Import packages. You can import additional packages, if you want

# Use the following solver
solver = COIN_CMD(threads=8, msg=False)

# Input file is called ./bakery.txt
input_filename = '/bakery.txt'

# Use solver defined above as a parameter of .solve() method.

# Open the file
with open(input_filename, 'r') as file:
    data = np.array([list(lines.split()) for lines in file if lines != '\n'], dtype = np.int16)
length = data.shape[0]
#
# ID PRE DLN BAK
#

# Choose the smallest M such that it works for the big-M method
M = max(data[i][2] for i in range(length)) + 1

# Initialize the mixed LP
lp = LpProblem('bakery', sense = LpMinimize)

# Define the objective
max_time = LpVariable('minimum maximum time', cat = 'Continuous', lowBound=0)
lp += max_time, 'obj'

# Starting time variables and constraints
var_s = list()
for i in range(length):
    var_s.append(LpVariable(f's_{i}', cat = 'Continuous', lowBound=0))
    lp += var_s[i] >= data[i][1], ''
    lp += var_s[i] <= (data[i][2] - data[i][3]), ''
    lp += max_time >= (var_s[i] + data[i][3]), ''

# Auxiliary binary variables for big-M method, along with their constraints
for i in range(length):
    for j in range(0, i):
        temp = LpVariable(f'z_{i},{j}', cat = 'Binary')
        #si >= ej
        lp += var_s[i] >= var_s[j] + data[j][3] - M + M * temp, '' #True if z=1
        #sj >= ei
        lp += var_s[j] >= var_s[i] + data[i][3] - M * temp, '' #True if z=0

# Solve the mixed LP
#print(lp)
print(lp.solve(solver))

# Get the solution
retval = dict()
for i in range(length):
    retval[f's_{i}'] = var_s[i].value()

# retval should be a dictionary such that retval['s_i'] is the starting
# time of pastry i

print(retval)
print(max_time.value())
